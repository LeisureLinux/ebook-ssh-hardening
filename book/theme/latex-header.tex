% ============================================================
% ebook-ssh-hardening — LaTeX header (XeLaTeX)
% Loaded via pandoc --include-in-header
% Purpose:
%   1. Render chapter headings as「第 N 章」instead of "Chapter N"
%   2. Configure CJK fonts so that both main text and code blocks
%      can display Chinese characters without tofu (□).
% ============================================================

% ---- 1. Chinese chapter prefix ----
% Default LaTeX book class uses \chaptername for "Chapter N".
% We redefine it to render "第 N 章".
\usepackage{xeCJK}
\setCJKmainfont{Noto Serif CJK SC}
\setCJKsansfont{Noto Sans CJK SC}
\setCJKmonofont{Noto Sans Mono CJK SC}

% Add a thin CJK-aware setting for chapter headings
\renewcommand{\chaptername}{第}
\renewcommand{\thechapter}{\arabic{chapter}}
% Make chapter headings render as "第 1 章 <title>" in TOC and body
% (book class default is "Chapter 1"; the format is hard-coded in
% the .cls file so we patch it.)
\makeatletter
\def\@makechapterhead#1{%
  \vspace*{50\p@}%
  {\parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \if@mainmatter
        \huge\bfseries 第 \thechapter 章 \par\nobreak
        \vskip 20\p@
      \fi
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak
    \vskip 40\p@
  }}
\def\@makeschapterhead#1{%
  \vspace*{50\p@}%
  {\parindent \z@ \raggedright
    \normalfont
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak
    \vskip 40\p@
  }}
\makeatother

% ---- 2. CJK-aware code/verbatim blocks ----
% Make Verbatim / fancyvrb use CJK monospace so Chinese characters
% inside bash/shell code blocks render correctly.
\IfFileExists{fancyvrb.sty}{%
  \DefineVerbatimEnvironment{Highlighting}{Verbatim}{commandchars=\\\{\},fontsize=\small}%
}{}

% Force monospace text to fall back to a CJK-capable mono font.
% We bind \ttfamily to a fontspec declaration so any \texttt{} or
% \verb command can render Chinese characters.
\usepackage{fontspec}
\setmonofont{Noto Sans Mono CJK SC}[
  Scale=0.9,
  AutoFakeSlant=0.15,
  BoldFont=Noto Sans Mono CJK SC Bold
]

% ---- 3. Page geometry tweaks ----
\usepackage{geometry}
\geometry{
  a4paper,
  margin=2.5cm,
  top=2.8cm,
  bottom=2.8cm
}

% ---- 4. Table of contents depth ----
\setcounter{tocdepth}{2}     % show sections (1.1, 1.2) but not subsections (1.1.1)
\setcounter{secnumdepth}{3}  % but number subsections in the body

% ---- 5. Headers/footers ----
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE]{\thepage\quad\textit{\leftmark}}
\fancyhead[RO]{\textit{\rightmark}\quad\thepage}
\renewcommand{\chaptermark}[1]{\markboth{第 \thechapter 章\quad #1}{}}
\renewcommand{\sectionmark}[1]{\markright{\thesection\quad #1}}

% ---- 6. Listings / code block style ----
\usepackage{listings}
\definecolor{codegreen}{rgb}{0,0.6,0}
\definecolor{codegray}{rgb}{0.5,0.5,0.5}
\definecolor{codebg}{rgb}{0.97,0.97,0.97}
\lstset{
  basicstyle=\ttfamily\small,
  commentstyle=\color{codegreen},
  keywordstyle=\color{blue},
  stringstyle=\color{red!70!black},
  backgroundcolor=\color{codebg},
  frame=single,
  framerule=0pt,
  framesep=6pt,
  xleftmargin=8pt,
  breaklines=true,
  showstringspaces=false
}
